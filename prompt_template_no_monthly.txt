quality_factor = roe.ffill().shift(1)

# 3. Combine factors
combined = momentum * 0.3 + growth_factor * 0.3 + value_factor * 0.2 + quality_factor * 0.2

# 4. Filter

price_filter = close.shift(1) > 10
filters = liquidity_filter & price_filter

# 5. Select
position = combined[filters].is_largest(10)

# 6. Backtest
report = sim(position, resample="Q", upload=False, stop_loss=0.08)
